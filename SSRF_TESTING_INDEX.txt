Quick Command Reference:
────────────────────────

# Option 1: Automated (easiest)
chmod +x test_ssrf_demo.sh
./test_ssrf_demo.sh

# Option 2: Manual
cd demo_vuln_app && python app.py --port 8000 & && cd ..
sleep 3
python main.py --scan ssrf -f demo_vuln_app/urls_ssrf.txt --listener http://127.0.0.1:5000 --wait 30 --threads 2

# Option 3: With recon
python main.py --recon -u http://127.0.0.1:8000 --scan ssrf --listener http://127.0.0.1:5000 --wait 30

# View results
cat bssrf/output/findings_ssrf.json | jq .
curl http://127.0.0.1:5000/api/callbacks | jq .
